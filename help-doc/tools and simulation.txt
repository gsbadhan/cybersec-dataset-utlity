---
cd ~/Downloads/cybersec-dataset/darpa/1999/week2
for day in monday tuesday wednesday thursday friday
do
  gunzip $day/inside.tcpdump.gz
  gunzip $day/outside.tcpdump.gz
  gunzip $day/pascal.bsm.gz
  tcpdump -nn -r $day/inside.tcpdump > $day/inside.tcpdump.log
  tcpdump -nn -r $day/outside.tcpdump > $day/outside.tcpdump.log
  tcpdump -nn -r $day/pascal.bsm > $day/pascal.bsm.log
done
